        <script>
            let currentInput = '';

            function updateDisplay() {
                document.getElementById('display').value = currentInput;
            }

            function appendNumber(number) {
                currentInput += number;
                updateDisplay();
            }

            function appendDot() {
                if (!currentInput.includes('.')) {
                    currentInput += '.';
                    updateDisplay();
                }
            }

            function clearDisplay() {
                currentInput = '';
                updateDisplay();
            }

            function calculate() {
                try {
                    currentInput = eval(currentInput).toString();
                    updateDisplay();
                } catch (error) {
                    currentInput = 'Error';
                    updateDisplay();
                }
            }
        </script>
